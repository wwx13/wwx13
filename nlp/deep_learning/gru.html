# 梯度裁剪


def grad_clipping(params, theta, device):
    norm = torch.tensor([0.0], device=device)
    for param in params:
        norm += (param.grad.data ** 2).sum()
    norm = norm.sqrt().item()
    if norm > theta:
        for param in params:
            param.grad.data *= (theta / norm)

import tensorflow as tf

class GRU():
    def __init__(self, hidden, out_dim, init_state, batch_size, scope_name="GRU"):
        self.hidden = hidden
        self.out_dim = out_dim
        self.scope_name = scope_name
        if not init_state:
            self.init_state = tf.random.truncated_normal([batch_size, hidden])
        else:
            self.init_state = init_state
            
    def get_variable(self, name_feature, name_hidden, shape_feature,
        shape_hidden, bias_feature=True, bias_hidden=True, get_out_param=False,
        initializer=tf.random_normal_initializer(stddev=0.1)
        ):
        feature_param = tf.get_variable(
            name_feature, shape=shape_feature, initializer=initializer
        )
        feature_bias = None
        if bias_feature:
            feature_bias = tf.get_variable(
                name_feature+"bias", shape=shape_feature[-1:],initializer=initializer
            )
        
        hidden_param = tf.get_variable(
            name_hidden, shape=shape_hidden,initializer=initializer
        )
        hidden_bias = None
        if bias_hidden:
            hidden_bias = tf.get_variable(
                name_hidden+"bias", shape=shape_hidden[-1:],initializer=initializer
            )
        out_param = None
        if get_out_param:
            out_param = tf.get_variable(
                "output", shape=[self.hidden, self.out_dim],initializer=initializer
            )
   
        return (feature_param, feature_bias), (hidden_param, hidden_bias), out_param
    
    def forward_op(self, state, temp_input_feature, params, activation_func=None):
        feature_param, feature_bias, hidden_param, hidden_bias, out_param = (params)
        
        new_state = tf.add(tf.matmul(state, hidden_param), hidden_bias)
        feature = tf.add(tf.matmul(temp_input_feature, feature_param), feature_bias)
        
        if activation_func:
            return  activation_func(new_state), activation_func(feature)
        return new_state, feature
        
    def run(self, input_feature, do_return_all_state):
        # [batch*embed_size] list of (batch * embed_size)
        state = self.init_state
        all_state = []
        all_output = []
        for idx, val in enumerate(input_feature):
            with tf.get_variable_scope(self.scope_name, reuse=tf.AUTO_REUSE):
                (W_f, W_f_b), (W_h, W_h_b), out_param = self.get_variable(
                    "gru_feature_reset", [input_feature[0].shape[-1], self.hidden],  
                    "gru_hidden_reset", [self.hidden, self.hidden],
                    bias_feature=True, bias_hidden=True, get_out_param=False
                )
                reset_gate = self.forward_op(
                    state=state, temp_input_feature=val, 
                    params=(W_f, W_f_b, W_h, W_h_b), activation_func=tf.sigmoid
                )
                
                (W_f2, W_f_b2), (W_h2, W_h_b2), out_param2 = self.get_variable(
                    "gru_feature_update", [input_feature[0].shape[-1], self.hidden],  
                    "gru_hidden_update", [self.hidden, self.hidden],
                    bias_feature=True, bias_hidden=True, get_out_param=False
                )
                update_gate = self.forward_op(
                    state=state, temp_input_feature=val, 
                    params=(W_f2, W_f_b2, W_h2, W_h_b2), activation_func=tf.sigmoid
                )
        
                (W_hat, W_hat_b), (W_hat_hidden, W_hidden_b), out_param_temp = self.get_variable(
                    "gru_feature_hat", [input_feature[0].shape[-1], self.hidden],  
                    "gru_hidden_hat", [self.hidden, self.hidden],
                    bias_feature=True, bias_hidden=False, get_out_param=True
                )         
                         
                H_hat = tf.tanh(
                    tf.add(
                        tf.matmul(val, W_hat) + reset_gate * tf.matmul(state, W_hat_hidden),
                        W_hat_b
                    )
                )
                
                state = update_gate * state + (1-update_gate)* H_hat
                if do_return_all_state:
                    all_state.append(state)
                    
                output = tf.matmul(state, out_param_temp)
                all_output.append(output)
                
        return all_output, all_state

import tensorflow as tf

class LSTM():
    def __init__(self, hidden, out_dim, init_state, batch_size, scope_name="LSTM"):
        self.hidden = hidden
        self.out_dim = out_dim
        self.scope_name = scope_name
        if not init_state:
            self.init_state = tf.random.truncated_normal([batch_size, hidden])
        else:
            self.init_state = init_state
        self.cell = tf.random.truncated_normal([batch_size, hidden])
            
    def get_variable(self, name_feature, name_hidden, shape_feature,
        shape_hidden, bias_feature=True, bias_hidden=True, get_out_param=False,
        initializer=tf.random_normal_initializer(stddev=0.1)
        ):
        feature_param = tf.get_variable(
            name_feature, shape=shape_feature, initializer=initializer
        )
        feature_bias = None
        if bias_feature:
            feature_bias = tf.get_variable(
                name_feature+"bias", shape=shape_feature[-1:],initializer=initializer
            )
        
        hidden_param = tf.get_variable(
            name_hidden, shape=shape_hidden,initializer=initializer
        )
        hidden_bias = None
        if bias_hidden:
            hidden_bias = tf.get_variable(
                name_hidden+"bias", shape=shape_hidden[-1:],initializer=initializer
            )
        out_param = None
        if get_out_param:
            out_param = tf.get_variable(
                "output", shape=[self.hidden, self.out_dim],initializer=initializer
            )
   
        return (feature_param, feature_bias), (hidden_param, hidden_bias), out_param
    
    def forward_op(self, state, temp_input_feature, params, activation_func=None):
        feature_param, feature_bias, hidden_param, hidden_bias, out_param = (params)
        
        new_state = tf.add(tf.matmul(state, hidden_param), hidden_bias)
        feature = tf.add(tf.matmul(temp_input_feature, feature_param), feature_bias)
        
        if activation_func:
            return  activation_func(new_state), activation_func(feature)
        return new_state, feature
        
    def run(self, input_feature, do_return_all_state):
        # [batch*embed_size] list of (batch * embed_size)
        state = self.init_state
        cell = self.cell
        all_state = []
        all_output = []
        for idx, val in enumerate(input_feature):
            with tf.get_variable_scope(self.scope_name, reuse=tf.AUTO_REUSE):
                (W_f, W_f_b), (W_h, W_h_b), out_param = self.get_variable(
                    "lstm_feature_input", [input_feature[0].shape[-1], self.hidden],  
                    "lstm_hidden_input", [self.hidden, self.hidden],
                    bias_feature=True, bias_hidden=True, get_out_param=False
                )
                input_gate = self.forward_op(
                    state=state, temp_input_feature=val, 
                    params=(W_f, W_f_b, W_h, W_h_b), activation_func=tf.sigmoid
                )
                
                (W_f2, W_f_b2), (W_h2, W_h_b2), out_param2 = self.get_variable(
                    "lstm_feature_forget", [input_feature[0].shape[-1], self.hidden],  
                    "lstm_hidden_forget", [self.hidden, self.hidden],
                    bias_feature=True, bias_hidden=True, get_out_param=False
                )
                forget_gate = self.forward_op(
                    state=state, temp_input_feature=val, 
                    params=(W_f2, W_f_b2, W_h2, W_h_b2), activation_func=tf.sigmoid
                )
        
                (W_f3, W_f_b3), (W_h3, W_h_b3), out_param3 = self.get_variable(
                    "lstm_feature_output", [input_feature[0].shape[-1], self.hidden],  
                    "lstm_hidden_output", [self.hidden, self.hidden],
                    bias_feature=True, bias_hidden=True, get_out_param=False
                )
                output_gate = self.forward_op(
                    state=state, temp_input_feature=val, 
                    params=(W_f3, W_f_b3, W_h3, W_h_b3), activation_func=tf.sigmoid
                )
                
        
                (W_hat, W_hat_b), (W_hat_hidden, W_hidden_b), out_param_temp = self.get_variable(
                    "lstm_feature_hat", [input_feature[0].shape[-1], self.hidden],  
                    "lstm_hidden_hat", [self.hidden, self.hidden],
                    bias_feature=True, bias_hidden=False, get_out_param=True
                )         
                         
                cell_hat = tf.tanh(
                    tf.add(
                        tf.matmul(val, W_hat) +  tf.matmul(state, W_hat_hidden),
                        W_hat_b
                    )
                )
                
                cell = forget_gate * cell + input_gate * cell_hat
                
                state = output_gate * tf.tanh(cell)
                if do_return_all_state:
                    all_state.append(state)
                    
                output = tf.matmul(state, out_param_temp)
                all_output.append(output)
                
        return all_output, all_state
