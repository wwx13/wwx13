# 多头实现

# 包含attetion layer和dense层
import tensorflow as tf
import math

def reshape_head(feature, n_head, batch, seq_len):
    return tf.reshape(feature, [batch, seq_len, n_head, -1])

def multi_head_layer(feature, n_head, drop_prob, pad_mask=None):
    # feature: batch * seq * hidden
    # pad_mask: batch * seq* seq, 0 is pading
    batch_size, seq_len, hidden_size = tf.shape(feature)
    query = tf.layer.Dense(feature, hidden_size, name="Query dense")
    key = tf.layer.Dense(feature, hidden_size, name="Key dense")
    value = tf.layer.Dense(feature, hidden_size, name="Value dense")
    
    query = reshape_head(query, n_head, batch_size, seq_len)
    key = reshape_head(key, n_head, batch_size, seq_len)
    value = reshape_head(value, batch_size, seq_len)
    
    att_score = tf.multiply(
        tf.matmul(query, key, transpose_b=True), 
        1 / tf.cast(math.sqrt(tf.shape(query)[-1]), tf.float32)
    )
    # batch * n_head * seq * seq    
    if pad_mask is not None:
        att_score + tf.epand_dims((1 - pad_mask) * -1000, 1)
    att_score = tf.nn.softmax(att_score, -1)
    att_score = tf.nn.dropout(att_score, 1-drop_prob)
    context = tf.matmul(att_score, value)
    
    return context
