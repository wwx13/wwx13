# 拓扑排序

  1. 思想： 给定有向图，我们按照依赖关系给出排序，使得被指向点总是在指向点之后。 算法通过维护入度为0的点的集合完成排序，初始时把入度为0的点放入集合，每次pop出一个点，加入排序列表， 并把该点指向的边都删除，确定那些被指向的点，有没有入度为0的也加入集合。

class Solution:
    def longestIncreasingPath(self, matrix: List[List[int]]) -> int:

        visited = {}  # key pos, val is longgest continue len

        queue = []

        def is_valid(pos, row, col):
            if (-1<pos[0]<row) and (-1<pos[1]<col):
                return True
            return False

        def is_connect(matrix, row, col, pos, adj_pos):
            if is_valid(adj_pos, row, col):
                if  matrix[pos[0]][pos[1]] > matrix[adj_pos[0]][adj_pos[1]]:
                    return True
            return False
            # return True

        row = len(matrix)
        col = len(matrix[0])
        node_with_num = {}
        for i in range(row):
            for j in range(col):
                pos = [i,j]
                pos_key = str(pos[0])+"@"+str(pos[1])
                direct = [[1,0],[-1,0],[0,1],[0,-1]]
                num = 0
                for d in direct:
                  new_pos = [pos[0]+d[0],pos[1]+d[1]]
                  if is_connect(matrix, row, col, pos, new_pos):
                      num += 1
                      
                if num > 0:
                    node_with_num[pos_key] = num
                    continue
                queue.append(pos)
        direct = [[1,0],[-1,0],[0,1],[0,-1]]
        # print(queue)
        while queue:
            temp = queue.pop(0)
            temp_connect_len = 1
            temp_key = str(temp[0])+"@"+str(temp[1])
            # print(temp)
            for d in direct:
                new_pos = [temp[0]+d[0],temp[1]+d[1]]
                new_pos_key = str(new_pos[0])+"@"+str(new_pos[1])
                # print("--", new_pos)
                if new_pos_key in visited:
                    if is_connect(matrix, row, col, temp, new_pos):
                        temp_connect_len = max(visited[new_pos_key] + 1, temp_connect_len)
                elif is_valid(new_pos, row, col) and is_connect(matrix, row, col, new_pos, temp):
                    node_with_num[new_pos_key] -= 1
                    if node_with_num[new_pos_key] == 0:
                        queue.append(new_pos)
                        del node_with_num[new_pos_key]
            visited[temp_key] = temp_connect_len
        return max(visited.values())
