class Solution:
    def minDistance(self, word1: str, word2: str) -> int:

        visited = {}

        left = 0
        right = 0
        len1 = len(word1)
        len2 = len(word2)
        
        for left in range(len1+1):
            for right in range(len2+1):
                temp_key = str(left)+"@"+str(right)
                if left*right == 0:
                    visited[temp_key] = abs(right - left)
                else:
                    k = str(left-1)+"@"+str(right-1)
                    a1 = visited[k] + 2*int(word2[right-1] != word1[left-1])  # 相同保留,不同同时删除 
                    a4 = visited[k]+1  #替换一个字符

                    # 删除,插入一个
                    k= str(left-1)+"@" + str(right)
                    a2 = visited[k] + 1
                    k= str(left)+"@" + str(right-1)
                    a3 = visited[k]+1
                    visited[temp_key] = min(a1,a2,a3,a4)
                    
        return visited[str(len1)+"@"+str(len2)]

class Solution:
    def longestIncreasingPath(self, matrix: List[List[int]]) -> int:

        row_num = len(matrix)
        col_num = len(matrix[0])
        visit = {}
        def dfs(row, col):
            direct = [[-1,0],[1,0],[0,1],[0,-1]]
            maxlen=1
            key = str(row)+"@"+str(col)
            if key in visit:
                return visit[key]
            for d in direct:
                new_row = row+d[0]
                new_col = col + d[1]
                if 0<=new_row<row_num and 0<=new_col<col_num and matrix[new_row][new_col]<matrix[row][col] :
                    maxlen = max(dfs(row+d[0],col+d[1]) + 1, maxlen)
            visit[key]=maxlen
            return maxlen
        for i in range(row_num):
            for j in range(col_num):
                dfs(i,j) 
    
        return max(visit.values())
