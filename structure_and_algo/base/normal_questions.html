def query_sqart(a, st, ed, eps):
    if st > ed:
        raise Exception("错误的开始和结束")

    mid = float((st + ed) / 2)
    print(mid)
    sqrted_mid = mid ** 2
    if abs(sqrted_mid - a) < eps:
        return truncate(mid)
    elif sqrted_mid > a:
        return query_sqart(a, st, mid, eps)
    return query_sqart(a, mid, ed, eps)


def truncate(num):
    num = float(num)
    left, right = str(num).split(".")
    return float(left + "." + right[:2])

print(query_sqart(2, 0, 2, eps=0.0001))

def eval_loss(x, a):
    if x**2 > a:
        neg_gd = -2 * x
        return x**2 - a, neg_gd
    neg_gd = 2 ** x
    return -(x**2) + a, neg_gd


def train(init_x, a, lr, eps):
    loss = 999
    x = init_x
    while loss >= eps:
        loss, neg_grad = eval_loss(x, a)
        x = sgd(x, lr, neg_grad)
        print("temp param: {}, loss: {}".format(x, loss))
    return x


def sgd(param, lr, grad):
    param = param + lr * grad
    return param

train(100, 2, 0.01, 0.001)
